from __future__ import annotations
import os, requests

TOKEN_URL = "https://entreprise.francetravail.fr/connexion/oauth2/access_token"
SEARCH_URL = "https://api.francetravail.io/partenaire/offresdemploi/v2/offres/search"
SCOPE = "api_offresdemploiv2 o2dsoffre"

class FranceTravailError(RuntimeError): pass

def get_access_token():
    cid=os.environ.get("FRANCE_TRAVAIL_CLIENT_ID")
    sec=os.environ.get("FRANCE_TRAVAIL_CLIENT_SECRET")
    if not cid or not sec: raise FranceTravailError("Secrets France Travail manquants.")
    r=requests.post(TOKEN_URL, params={"realm":"/partenaire"}, data={
        "grant_type":"client_credentials","client_id":cid,"client_secret":sec,"scope":SCOPE
    }, timeout=30)
    if not r.ok: raise FranceTravailError(f"OAuth échoué {r.status_code}: {r.text[:500]}")
    return r.json()["access_token"]

def search_offers(token, keyword, latitude, longitude, radius_km, max_results=100):
    end=min(max(max_results-1,0),149)
    r=requests.get(SEARCH_URL, headers={"Authorization":f"Bearer {token}","Accept":"application/json"},
        params={"motsCles":keyword,"latitude":latitude,"longitude":longitude,
                "distance":radius_km,"range":f"0-{end}","sort":"1"}, timeout=45)
    if r.status_code==204: return []
    if not r.ok: raise FranceTravailError(f"Recherche échouée {r.status_code}: {r.text[:500]}")
    return r.json().get("resultats",[])
